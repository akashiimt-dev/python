function handleFileChange(e) {
  const filesArray = Array.from(e.target.files); // Convert FileList to an array

  // Function to rename a file by its original name
  function renameFile(oldName, newName) {
    const fileIndex = filesArray.findIndex(file => file.name === oldName);
    if (fileIndex !== -1) {
      const oldFile = filesArray[fileIndex];
      const newFile = new File([oldFile], newName, {
        type: oldFile.type,
        lastModified: oldFile.lastModified
      });

      // Replace the old file with the new file
      filesArray[fileIndex] = newFile;
      console.log(`Renamed "${oldName}" to "${newName}"`);
    } else {
      console.log(`File "${oldName}" not found.`);
    }
  }

  // Example: Rename a file
  renameFile("file1.txt", "new_file1.txt");

  // Logging the renamed files
  console.log(filesArray.map(file => file.name));
}

// Example usage in an HTML input
// <input type="file" multiple onchange="handleFileChange(event)" />
