const array = [
  { key: 'value1', otherKey: 1 },
  { key: 2, otherKey: 'data2' },
  { key: 'value3', otherKey: 3 }
];

// Set default values
array.forEach(obj => {
  if (typeof obj.key === 'string') {
    obj.key = ''; // Default to blank string
  } else if (typeof obj.key === 'number') {
    obj.key = 0; // Default to 0
  } else {
    obj.key = []; // Default to empty array for other types
  }
});

console.log(array);
